const searchInput=document.getElementById('searchInput'); const categorySelect=document.getElementById('categorySelect'); const buttons=[...document.querySelectorAll('[data-filter]')]; const cards=[...document.querySelectorAll('[data-article]')]; const empty=document.getElementById('emptyState'); let selected=new URLSearchParams(location.search).get('category')||'all'; function sync(){ if(categorySelect) categorySelect.value=selected; buttons.forEach(b=>b.classList.toggle('active',b.dataset.filter===selected)); } function applyFilter(){ const q=(searchInput?.value||'').toLowerCase().trim(); let visible=0; cards.forEach(card=>{ const text=card.textContent.toLowerCase(); const category=card.dataset.category; const show=(selected==='all'||category===selected)&&(!q||text.includes(q)); card.style.display=show?'block':'none'; if(show) visible++; }); if(empty) empty.style.display=visible?'none':'block'; } buttons.forEach(btn=>btn.addEventListener('click',()=>{selected=btn.dataset.filter;sync();applyFilter();})); categorySelect?.addEventListener('change',()=>{selected=categorySelect.value;sync();applyFilter();}); searchInput?.addEventListener('input',applyFilter); sync();applyFilter();
